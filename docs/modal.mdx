## Example

<Playground>
  {() => {
    class Example extends React.Component {
      constructor() {
        this.state = {
          visible: false,
        }
        this.toggleVisible = () => {
          this.setState(state => ({
            visible: !state.visible,
          }))
        }
      }

      render() {
        return (
          <div>
            <Button onClick={this.toggleVisible}>Toggle</Button>
            <Modal
              visible={this.state.visible}
              body={<div>Woohoo, you're reading this text in a modal!</div>}
              onFinish={this.toggleVisible}
              onCancel={this.toggleVisible}
            />
          </div>
        )
      }
    }

    return <Example />

}}

</Playground>

## Confirm

<Playground>
  {() => {
    class Example extends React.Component {
      constructor() {
        this.state = {
          result: '',
        }
      }

      render() {
        return (
          <React.Fragment>
            <Button
              onClick={() => {
                Modal.confirm({
                  title: 'Confirm',
                  content: "Are you sure",
                  onFinish: ()=> {
                    this.setState({ result: 'finish' })
                  },
                  onCancel:() =>{
                    this.setState({ result: 'cancel' })
                  },
                })
              }
            }>Confirm</Button>{' '}
            <span>
              {this.state.result ? `You click ` + this.state.result : 'Click the button'}
            </span>
          </React.Fragment>
        )
      }
    }

    return <Example />

}}

</Playground>
