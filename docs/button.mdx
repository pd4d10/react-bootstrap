## Loading state

<Playground>
  <Provider state={{ loading: false, timeout: 5 }}>
    {({ state, setState }) => (
      <React.Fragment>
        <Button loading={true} disabled>Loading...</Button>{' '}
        <Button
          loading={state.loading}
          disabled={state.loading}
          onClick={() => { 
            setState({ loading: true })
            setTimeout(() => {
              setState({ loading: false })
            }, 5000)
          }}
        >Click me</Button>
      </React.Fragment>
    )}
  </Provider>
</Playground>
