## Loading state

<Playground>
  {() => {
    class Example extends React.Component {
      constructor() {
        this.state = { loading: false }
      }
      render() {
        return (
          <>
            <Button loading={true} disabled>
              Loading...
            </Button>{' '}
            <Button
              loading={this.state.loading}
              disabled={this.state.loading}
              onClick={() => {
                this.setState({ loading: true })
                setTimeout(() => {
                  this.setState({ loading: false })
                }, 2000)
              }}
            >
              Trigger loading
            </Button>
          </>
        )
      }
    }
    return <Example />
  }}
</Playground>
